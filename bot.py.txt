async def send_question(chat_id, index):

        q = questions[index]
        text = f"‚ùì –°–∞–≤–æ–ª {index + 1} –∞–∑ {len(questions)}:\n\n{q['q']}\n\n"

        await bot.send_message(chat_id, text)

    else:

# 15 —Å–æ–Ω–∏—è –º”Ø“≥–ª–∞—Ç

        await bot.send_message(chat_id, "‚åõ –í–∞“õ—Ç–∞—Ç –≥—É–∑–∞—à—Ç!")
        user_state[chat_id] = index + 1
        await send_question(chat_id, index + 1)

                await message.answer("‚ùå –¢–µ—Å—Ç “õ–∞—Ç—ä —à—É–¥ –±–æ —Å–∞–±–∞–±–∏ –∞–∑ 80% –∫–∞–º –≥–∏—Ä–∏—Ñ—Ç–∞–Ω. –ú–µ—Ç–∞–≤–æ–Ω”£ –±–∞—ä–¥—Ç–∞—Ä –±–æ–∑ –∫”Ø—à–∏—à –∫—É–Ω”£. /start")
                user_state.pop(chat_id, None)
                user_score.pop(chat_id, None)
                return
            else:
                await message.answer("‚úÖ –û—Ñ–∞—Ä–∏–Ω! –ë–∞ –¥–∞–≤—Ä–∏ –Ω–∞–≤–±–∞—Ç–∏–∏ —Å–∞–≤–æ–ª“≥–æ –º–µ–≥—É–∑–∞—Ä–µ–º!")
                await send_question(chat_id, index + 1)
                return

        # –ê–≥–∞—Ä “≥–∞–Ω”Ø–∑ 10-—Ç–æ –ø—É—Ä—Ä–∞ –Ω–∞—à—É–¥–∞–∞—Å—Ç
        await send_question(chat_id, index + 1)
    # –ë–∞ —Å–∞–≤–æ–ª–∏ –Ω–∞–≤–±–∞—Ç”£ –º–µ–≥—É–∑–∞—Ä–∞–¥
    user_state[chat_id] = index + 1
    await send_question(chat_id, index + 1)
